	function onSelectTab(e){
		e.preventDefault();

		t.parentsUntil('.tab-menu-4:first').filter('li').each(function(){

		$("#tab0"+id).show();
		$("#btn0"+id).show();
	}
	tab.find('li>a').on("click", onSelectTab);
